import React, { useContext, useState } from 'react';
import { Container, Table, Button, Modal } from 'react-bootstrap';
import { CartContext } from './CartContext';

const Carrito = () => {
  const { carrito, setCarrito } = useContext(CartContext);

  const [showModal, setShowModal] = useState(false);
  const [totalProductos, setTotalProductos] = useState(0);
  const [totalPago, setTotalPago] = useState(0);

  // --- ELIMINAR UN PRODUCTO ---
  const eliminarDelCarrito = (id) => {
    setCarrito(prev => prev.filter(producto => producto.id !== id));
  };

  // --- VACÍAR CARRITO ---
  const vaciarCarrito = () => {
    setCarrito([]);
  };

  const total = carrito.reduce((acc, item) => acc + Number(item.price) * item.cantidad, 0);

  // --- CONCRETAR COMPRA ---
  const concretarCompra = () => {
    const totalItems = carrito.reduce((acc, item) => acc + item.cantidad, 0);
    const totalAPagar = total;

    setTotalProductos(totalItems);
    setTotalPago(totalAPagar);

    // --- NUEVO: GUARDAR EN LOCALSTORAGE ---
    const comprasGuardadas = JSON.parse(localStorage.getItem('compras')) || [];
    const nuevaCompra = {
      fecha: new Date().toISOString(),
      totalProductos: totalItems,
      totalPago: totalAPagar,
      items: carrito
    };
    localStorage.setItem('compras', JSON.stringify([...comprasGuardadas, nuevaCompra]));

    setShowModal(true);
  };

  // --- CERRAR MODAL ---
  const cerrarModal = () => {
    setShowModal(false);
    setCarrito([]); // vaciar carrito
  };

  if (carrito.length === 0) {
    return (
      <Container className="mt-4">
        <h3>Tu carrito está vacío</h3>
      </Container>
    );
  }

  return (
    <Container className="mt-4">
      <h3>Carrito de compras</h3>

      {/* --- NUEVO: BOTÓN VACÍAR CARRITO --- */}
      <div className="d-flex justify-content-end mb-2">
        <Button variant="outline-danger" size="sm" onClick={vaciarCarrito}>
          Vaciar carrito
        </Button>
      </div>

      <Table striped bordered hover responsive className="mt-3">
        <thead>
          <tr>
            <th>Producto</th>
            <th>Precio unitario</th>
            <th>Cantidad</th>
            <th>Total</th>
            <th></th>
          </tr>
        </thead>
        <tbody>
          {carrito.map((item) => (
            <tr key={item.id}>
              <td>{item.title}</td>
              <td>${Number(item.price).toFixed(2)}</td>
              <td>{item.cantidad}</td>
              <td>${(Number(item.price) * item.cantidad).toFixed(2)}</td>
              <td>
                <Button
                  variant="danger"
                  size="sm"
                  onClick={() => eliminarDelCarrito(item.id)}
                >
                  Eliminar
                </Button>
              </td>
            </tr>
          ))}
        </tbody>
      </Table>

      <h5 className="text-end">Total a pagar: ${total.toFixed(2)}</h5>

      {/* BOTÓN CONCRETAR COMPRA */}
      <div className="text-end mt-3">
        <Button variant="success" onClick={concretarCompra}>
          Concretar compra
        </Button>
      </div>

      {/* MODAL DE CONFIRMACIÓN */}
      <Modal show={showModal} onHide={cerrarModal} centered>
        <Modal.Header closeButton>
          <Modal.Title>Compra realizada</Modal.Title>
        </Modal.Header>

        <Modal.Body>
          <p>
            Usted acaba de comprar <strong>{totalProductos}</strong> productos
            por un total de <strong>${totalPago.toFixed(2)}</strong>.
          </p>
          <h5>¡Gracias por su compra!</h5>
        </Modal.Body>

        <Modal.Footer>
          <Button variant="primary" onClick={cerrarModal}>
            Cerrar
          </Button>
        </Modal.Footer>
      </Modal>
    </Container>
  );
};

export default Carrito;
