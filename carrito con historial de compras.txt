import React, { useContext, useState, useEffect } from 'react';
import { Container, Table, Button, Modal } from 'react-bootstrap';
import { CartContext } from './CartContext';

const Carrito = () => {
  const { carrito, setCarrito } = useContext(CartContext);

  const [showModal, setShowModal] = useState(false);
  const [totalProductos, setTotalProductos] = useState(0);
  const [totalPago, setTotalPago] = useState(0);
  const [historial, setHistorial] = useState([]);

  // --- Cargar historial de localStorage al montar ---
  useEffect(() => {
    const comprasGuardadas = JSON.parse(localStorage.getItem('compras')) || [];
    setHistorial(comprasGuardadas);
  }, []);

  const eliminarDelCarrito = (id) => {
    setCarrito(prev => prev.filter(producto => producto.id !== id));
  };

  const vaciarCarrito = () => {
    setCarrito([]);
  };

  const total = carrito.reduce((acc, item) => acc + Number(item.price) * item.cantidad, 0);

  const concretarCompra = () => {
    const totalItems = carrito.reduce((acc, item) => acc + item.cantidad, 0);
    const totalAPagar = total;

    setTotalProductos(totalItems);
    setTotalPago(totalAPagar);

    // Guardar en localStorage
    const comprasGuardadas = JSON.parse(localStorage.getItem('compras')) || [];
    const nuevaCompra = {
      fecha: new Date().toISOString(),
      totalProductos: totalItems,
      totalPago: totalAPagar,
      items: carrito
    };
    const comprasActualizadas = [...comprasGuardadas, nuevaCompra];
    localStorage.setItem('compras', JSON.stringify(comprasActualizadas));
    setHistorial(comprasActualizadas);

    setShowModal(true);
  };

  const cerrarModal = () => {
    setShowModal(false);
    setCarrito([]);
  };

  return (
    <Container className="mt-4">
      <h3>Carrito de compras</h3>

      {carrito.length === 0 ? (
        <p>Tu carrito está vacío</p>
      ) : (
        <>
          {/* Botón Vaciar carrito */}
          <div className="d-flex justify-content-end mb-2">
            <Button variant="outline-danger" size="sm" onClick={vaciarCarrito}>
              Vaciar carrito
            </Button>
          </div>

          <Table striped bordered hover responsive className="mt-3">
            <thead>
              <tr>
                <th>Producto</th>
                <th>Precio unitario</th>
                <th>Cantidad</th>
                <th>Total</th>
                <th></th>
              </tr>
            </thead>
            <tbody>
              {carrito.map((item) => (
                <tr key={item.id}>
                  <td>{item.title}</td>
                  <td>${Number(item.price).toFixed(2)}</td>
                  <td>{item.cantidad}</td>
                  <td>${(Number(item.price) * item.cantidad).toFixed(2)}</td>
                  <td>
                    <Button
                      variant="danger"
                      size="sm"
                      onClick={() => eliminarDelCarrito(item.id)}
                    >
                      Eliminar
                    </Button>
                  </td>
                </tr>
              ))}
            </tbody>
          </Table>

          <h5 className="text-end">Total a pagar: ${total.toFixed(2)}</h5>

          <div className="text-end mt-3">
            <Button variant="success" onClick={concretarCompra}>
              Concretar compra
            </Button>
          </div>
        </>
      )}

      {/* Modal de confirmación */}
      <Modal show={showModal} onHide={cerrarModal} centered>
        <Modal.Header closeButton>
          <Modal.Title>Compra realizada</Modal.Title>
        </Modal.Header>

        <Modal.Body>
          <p>
            Usted acaba de comprar <strong>{totalProductos}</strong> productos
            por un total de <strong>${totalPago.toFixed(2)}</strong>.
          </p>
          <h5>¡Gracias por su compra!</h5>
        </Modal.Body>

        <Modal.Footer>
          <Button variant="primary" onClick={cerrarModal}>
            Cerrar
          </Button>
        </Modal.Footer>
      </Modal>

      {/* --- NUEVO: Historial de Compras --- */}
      <div className="mt-5">
        <h4>Historial de Compras</h4>
        {historial.length === 0 ? (
          <p>No hay compras registradas.</p>
        ) : (
          historial
            .slice() // copiar para no mutar
            .reverse() // mostrar la más reciente primero
            .map((compra, index) => (
              <div key={index} className="mb-4 p-3 border rounded">
                <p><strong>Fecha:</strong> {new Date(compra.fecha).toLocaleString()}</p>
                <p><strong>Total de productos:</strong> {compra.totalProductos}</p>
                <p><strong>Total pagado:</strong> ${compra.totalPago.toFixed(2)}</p>
                <Table striped bordered hover size="sm">
                  <thead>
                    <tr>
                      <th>Producto</th>
                      <th>Precio unitario</th>
                      <th>Cantidad</th>
                      <th>Total</th>
                    </tr>
                  </thead>
                  <tbody>
                    {compra.items.map((item) => (
                      <tr key={item.id}>
                        <td>{item.title}</td>
                        <td>${Number(item.price).toFixed(2)}</td>
                        <td>{item.cantidad}</td>
                        <td>${(Number(item.price) * item.cantidad).toFixed(2)}</td>
                      </tr>
                    ))}
                  </tbody>
                </Table>
              </div>
            ))
        )}
      </div>
    </Container>
  );
};

export default Carrito;
